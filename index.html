// ========== CONFIGURACIÓN ========== const GAS_URL = "https://script.google.com/macros/s/AKfycbw4fJGzFlbMb72uvZIZuxLDjVnqmjReJn77u-TCKOcF9htxhcvth3jGb7afPHvxaen5/exec"; let allProducts = []; // Elementos DOM const searchInput = document.getElementById('searchInput'); const filterLinea = document.getElementById('filterLinea'); const filterSubLinea = document.getElementById('filterSubLinea'); const filterGenero = document.getElementById('filterGenero'); const productsGrid = document.getElementById('productsGrid'); const loadingDiv = document.getElementById('loading'); const errorBox = document.getElementById('errorBox'); const debugInfo = document.getElementById('debugInfo'); function addLog(message, type = 'info') { console.log(`[${type.toUpperCase()}]`, message); if (debugInfo) { debugInfo.innerHTML += `\n${new Date().toLocaleTimeString()}: ${message}`; debugInfo.style.display = 'block'; if (type !== 'error') { setTimeout(() => { if (debugInfo) debugInfo.style.display = 'none'; }, 30000); } } } async function loadProducts() { addLog("Iniciando carga..."); loadingDiv.style.display = 'block'; productsGrid.innerHTML = ''; if (errorBox) errorBox.style.display = 'none'; try { const response = await fetch(GAS_URL); if (!response.ok) throw new Error(`HTTP ${response.status}`); const data = await response.json(); addLog(`Datos recibidos, productos: ${data.productos?.length || 0}`); if (data.error) throw new Error(data.error); if (!data.productos || !Array.isArray(data.productos)) throw new Error("Formato inválido"); allProducts = data.productos; if (allProducts.length === 0) { productsGrid.innerHTML = '
⚠️ No hay productos activos
'; return; } // CONVERTIR COD A STRING PARA EVITAR ERRORES allProducts = allProducts.map(p => ({ ...p, cod: String(p.cod) })); populateFilters(); renderProducts(); } catch (error) { addLog(`ERROR: ${error.message}`, "error"); if (errorBox) { errorBox.innerHTML = `❌ Error: ${error.message}
Ver consola F12`; errorBox.style.display = 'block'; } } finally { loadingDiv.style.display = 'none'; } } function populateFilters() { const lineas = new Set(); const subLineas = new Set(); const generos = new Set(); allProducts.forEach(p => { if (p.linea) lineas.add(p.linea); if (p.subLinea) subLineas.add(p.subLinea); if (p.genero) generos.add(p.genero); }); filterLinea.innerHTML = '' + Array.from(lineas).sort().map(l => ``).join(''); filterSubLinea.innerHTML = '' + Array.from(subLineas).sort().map(sl => ``).join(''); filterGenero.innerHTML = '' + Array.from(generos).sort().map(g => ``).join(''); } function filterProducts() { const searchTerm = searchInput.value.toLowerCase().trim(); const linea = filterLinea.value; const subLinea = filterSubLinea.value; const genero = filterGenero.value; return allProducts.filter(p => { let matchSearch = true; if (searchTerm) { matchSearch = (p.descripcion && p.descripcion.toLowerCase().includes(searchTerm)) || (p.cod && p.cod.toLowerCase().includes(searchTerm)) || (p.listaCodigos && p.listaCodigos.toLowerCase().includes(searchTerm)); } return matchSearch && (!linea || p.linea === linea) && (!subLinea || p.subLinea === subLinea) && (!genero || p.genero === genero); }); } function renderProducts() { const filtered = filterProducts(); if (filtered.length === 0) { productsGrid.innerHTML = '
😕 No se encontraron productos
'; return; } productsGrid.innerHTML = filtered.map(p => { const tieneFotos = (p.fotosSecundarias?.length || 0) + (p.imagenPrincipal ? 1 : 0); const imagenSrc = p.imagenPrincipal || 'https://via.placeholder.com/300x200?text=Sin+imagen'; // Usar data attribute en lugar de onclick con string para evitar problemas de escaping const codSafe = p.cod; // ya es string return `
${escapeHtml(p.descripcion || p.cod)}
Código: ${p.cod}
${(p.preciosMostrar || []).map(pr => ` ${escapeHtml(pr.label)}: $${pr.value?.toFixed(2)} `).join('')}
📸 ${tieneFotos} foto${tieneFotos !== 1 ? 's' : ''}
`; }).join(''); // Agregar event listeners a las tarjetas después de renderizar document.querySelectorAll('.product-card').forEach(card => { card.removeEventListener('click', cardClickHandler); card.addEventListener('click', cardClickHandler); }); } function cardClickHandler(e) { // Evitar que el click en elementos internos cierre el modal sin querer if (e.target.closest('.product-card')) { const cod = this.getAttribute('data-cod'); openGallery(cod); } } function openGallery(cod) { const product = allProducts.find(p => p.cod == cod); if (!product || !product.imagenPrincipal) return; const modal = document.getElementById('imageModal'); const modalImage = document.getElementById('modalImage'); modalImage.src = product.imagenPrincipal; modal.classList.add('active'); } function closeModal() { document.getElementById('imageModal').classList.remove('active'); } function escapeHtml(str) { if (!str) return ''; return str.replace(/[&<>]/g, function(m) { if (m === '&') return '&'; if (m === '<') return '<'; if (m === '>') return '>'; return m; }).replace(/[\uD800-\uDBFF][\uDC00-\uDFFF]/g, function(c) { return c; }); } function escapeAttr(str) { if (!str) return ''; return String(str).replace(/&/g, '&').replace(/"/g, '"'); } // Event listeners searchInput.addEventListener('input', () => renderProducts()); filterLinea.addEventListener('change', () => renderProducts()); filterSubLinea.addEventListener('change', () => renderProducts()); filterGenero.addEventListener('change', () => renderProducts()); document.addEventListener('keydown', (e) => { if (e.key === 'Escape') closeModal(); }); // Iniciar loadProducts();
